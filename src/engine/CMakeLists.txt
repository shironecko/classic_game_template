    engine_main.cpp
    window.cpp window.h
    pch.cpp pch.h external_libs.h
    system.cpp system.h
    assets.cpp assets.h
    clock.cpp clock.h
    ui_bindings.cpp ui_bindings.h

    NOMINMAX)
